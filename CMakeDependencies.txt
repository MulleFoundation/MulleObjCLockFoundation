if( NOT __MULLE_OBJC_CMAKE_DEPENDENCIES_TXT__)
   set( __MULLE_OBJC_CMAKE_DEPENDENCIES_TXT__ ON)

   message( STATUS "# Include MulleObjC CMakeDependencies.txt")

   include( ${DEPENDENCIES_DIR}/include/mulle_objc_runtime/CMakeDependencies.txt)
   include( ${DEPENDENCIES_DIR}/include/mulle_objc_list/CMakeDependencies.txt)
   include( ${DEPENDENCIES_DIR}/include/mulle_container/CMakeDependencies.txt)

   if( NOT MULLE_OBJC_LIBRARY)
      find_library( MULLE_OBJC_LIBRARY NAMES MulleObjC)
      message(STATUS "MULLE_OBJC_LIBRARY is ${MULLE_OBJC_LIBRARY}")
      set( OBJC_DEPENDENCY_LIBRARIES
         ${MULLE_OBJC_LIBRARY}
         ${OBJC_DEPENDENCY_LIBRARIES}
         CACHE INTERNAL "need to cache this"
      )
      set( OBJC_DEPENDENCY_NAMES
         MulleObjC
         ${OBJC_DEPENDENCY_NAMES}
         CACHE INTERNAL "need to cache this too"
      )
   endif()

   #
   # This library contains ___get_or_create_mulle_objc_universe and
   # the startup code to create the universe
   #
   if( NOT MULLE_OBJC_STARTUP_LIBRARY)
      find_library( MULLE_OBJC_STARTUP_LIBRARY NAMES MulleObjCStartup)
      message(STATUS "MULLE_OBJC_STARTUP_LIBRARY is ${MULLE_OBJC_STARTUP_LIBRARY}")

      # overwrite what came before
      set( OBJC_STARTUP_LIBRARY
         ${MULLE_OBJC_STARTUP_LIBRARY}
         CACHE INTERNAL "need to cache this"
      )
   endif()

   if( MSVC)
      set( MULLE_OBJC_DEFINITIONS
         ${MULLE_OBJC_RUNTIME_DEFINITIONS}
         ${MULLE_OBJC_LIST_DEFINITIONS}
         ${MULLE_CONTAINER_DEFINITIONS}
      )
      list( REMOVE_DUPLICATES MULLE_OBJC_DEFINITIONS)
      message( STATUS "MULLE_OBJC_DEFINITIONS = ${MULLE_OBJC_DEFINITIONS}")
   endif()

endif()
