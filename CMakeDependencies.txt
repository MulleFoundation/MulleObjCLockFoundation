if( NOT __MULLE_OBJC_CMAKE_DEPENDENCIES_TXT__)
   set( __MULLE_OBJC_CMAKE_DEPENDENCIES_TXT__ ON)

   message( STATUS "# Include MulleObjC CMakeDependencies.txt")

   include( ${DEPENDENCIES_DIR}/include/mulle_objc_runtime/CMakeDependencies.txt)

   if( NOT MULLE_CONTAINER_LIBRARY)
      find_library( MULLE_CONTAINER_LIBRARY NAMES mulle_container)
      message( STATUS "MULLE_CONTAINER_LIBRARY is ${MULLE_CONTAINER_LIBRARY}")
      set( C_DEPENDENCY_LIBRARIES
         ${MULLE_CONTAINER_LIBRARY}
         ${C_DEPENDENCY_LIBRARIES}
         CACHE INTERNAL "need to cache this"
      )
   endif()

   if( NOT MULLE_OBJC_RUNTIME_LIBRARY)
      find_library( MULLE_OBJC_RUNTIME_LIBRARY NAMES mulle_objc_runtime)
      message(STATUS "MULLE_OBJC_RUNTIME_LIBRARY is ${MULLE_OBJC_RUNTIME_LIBRARY}")
      set( C_DEPENDENCY_LIBRARIES
         ${MULLE_OBJC_RUNTIME_LIBRARY}
         ${C_DEPENDENCY_LIBRARIES}
         CACHE INTERNAL "need to cache this"
      )
   endif()

   if( NOT MULLE_OBJC_LIBRARY)
      find_library( MULLE_OBJC_LIBRARY NAMES MulleObjC)
      message(STATUS "MULLE_OBJC_LIBRARY is ${MULLE_OBJC_LIBRARY}")
      set( OBJC_DEPENDENCY_LIBRARIES
         ${MULLE_OBJC_LIBRARY}
         ${OBJC_DEPENDENCY_LIBRARIES}
         CACHE INTERNAL "need to cache this"
      )
      set( OBJC_DEPENDENCY_NAMES
         MulleObjC
         ${OBJC_DEPENDENCY_NAMES}
         CACHE INTERNAL "need to cache this too"
      )
   endif()

   message( STATUS "# Included MulleObjC CMakeDependencies.txt")

endif()
